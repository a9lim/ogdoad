# ===================================== v2.0 — lambdas, capture, application
@world integer 0

> abs := u ~ (u < 0 ? -u : u)
~ abs := u ↦ u < 0 ? -u : u

> abs@(-5)
= 5

> abs@7
= 7

> abs
= u ↦ u < 0 ? -u : u

> (u ↦ u⋅u)@10
= 100

> u ↦ u + 1
= u ↦ u + 1

# capture at definition is substitution — visible, and rebind-proof
> c := 5

> f := u ↦ c⋅u

> f
= u ↦ 5⋅u

> c := 7

> f@1
= 5

# a binder may shadow an ordinary binding (c is 7 here; the binder wins)
> g := c ↦ c + 1

> g
= c ↦ c + 1

> g@1
= 2

> sign := u ↦ (u < 0 ? -1 : (u = 0 ? 0 : 1))

> sign@0
= 0

> sign@4
= 1

> even2 := u ↦ u % 2 = 0
~ even2 := u ↦ u%2 = 0

> even2@4
= true

> even2@7
= false
